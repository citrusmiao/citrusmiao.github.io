# 表格封装

  1. 定义 CommonTable.vue 文件
<template>
  <div class="common-table">
    <el-table
      ref="table"
      v-loading="tableAttribute.loading"
      :border="tableAttribute.border"
      :fit="tableAttribute.fit"
      :stripe="tableAttribute.stripe"
      :height="tableAttribute.height"
      :show-header="tableAttribute.header"
      :data="tableData"
      :row-style="tableAttribute.rowStyle"
      :size="tableAttribute.size"
      style="width: 100%"
      tooltip-effect="light"
      @selection-change="handleSelectionChange"
    >
      <!-- 多选框 -->
      <el-table-column
        v-if="tableAttribute.selection"
        fixed
        type="selection"
        width="50"
        align="center"
      />

      <!-- 索引号 -->
      <el-table-column
        v-if="tableAttribute.index"
        fixed
        type="index"
        label="序号"
        width="50"
        align="center"
      />

      <!--数据列表-->
      <template v-for="(col, index) in tableCols">
        <el-table-column
          v-if="col.slot_name"
          :key="col.prop + 'slot' + index"
          :show-overflow-tooltip="!col.overflow"
          :prop="col.prop"
          :label="col.label"
          :min-width="col.minWidth"
          :width="col.width"
          :sortable="col.sortable"
          :fixed="col.fixed"
          :align="col.align || 'center'"
        >
          <template v-slot="scope">
            <slot :name="col.slot_name" :row="scope.row" />
          </template>
        </el-table-column>
        <el-table-column
          v-else
          :key="col.prop + 'col' + index"
          :show-overflow-tooltip="!col.overflow"
          :prop="col.prop"
          :label="col.label"
          :min-width="col.minWidth"
          :width="col.width"
          :sortable="col.sortable"
          :fixed="col.fixed"
          :align="col.align || 'center'"
        >
          <template v-slot="scope">
            <span>{{ scope.row[col.prop] || "-" }}</span>
          </template>
        </el-table-column>
      </template>
    </el-table>

    <!--分页组件-->
    <div v-if="pagination.total > 0" class="pagination-container">
      <el-pagination
        background
        :current-page="currentPage"
        :page-size="pageSize"
        :page-sizes="[10, 20, 50, 100, 200, 500]"
        layout="total, sizes, prev, pager, next, jumper"
        :total="pagination.total"
        @current-change="handleCurrentChange"
        @size-change="handleSizeChange"
      />
    </div>
  </div>
</template>

<script>
export default {
  name: "CommonTable",
  props: {
    tableAttribute: {
      type: Object,
      default: () => ({
        loading: false, // 加载
        loadingText: "", // 加载文字
        header: true, // 是否显示表头
        fit: true, // 列的宽度是否自撑开
        border: true, // 是否带有纵向边框
        stripe: true, // 是否为斑马纹 table
        height: undefined, // Table 的高度，默认为自动高度
        rowStyle: {}, // 行的样式
        selection: false, // 多选框
        index: true, // 序号
      }),
    },
    tableCols: { required: true, type: Array, default: () => [] },
    tableData: { type: Array, default: () => [] },
    pagination: {
      type: Object,
      default: () => ({ page: 1, pageSize: 10, total: 0 }),
    },
  },
  computed: {
    currentPage: {
      get() {
        return this.pagination.page;
      },
      set(val) {
        this.$emit("update:pagination.page", val);
      },
    },
    pageSize: {
      get() {
        return this.pagination.pageSize;
      },
      set(val) {
        this.$emit("update:pagination.pageSize", val);
      },
    },
  },
  methods: {
    handleSelectionChange(val) {
      this.$emit("handleSelectionChange", val);
    },
    handleCurrentChange(val) {
      this.$emit("handlePageChange", { page: val, pageSize: this.pageSize });
    },
    handleSizeChange(val) {
      this.$emit("handlePageChange", { page: this.currentPage, pageSize: val });
    },
  },
};
</script>

<template>
  <CommonTable
    :tableAttribute="tableAttribute"
    :tableCols="tableCols"
    :tableData="tableData"
    :pagination="pagination"
    @handlePageChange="handlePageChange"
  >
    <template v-slot:btn="data">
      <el-button type="primary" size="mini" @click="show(data.row)">按钮</el-button>
    </template>
  </CommonTable>
</template>
<script>
import CommonTable from "@/components/CommonTable";
export default {
  components: { CommonTable },
  data() {
    return {
      tableAttribute: { border: true, index: true },
      tableCols: [
        { prop: "name", label: "名称" },
        { prop: "policyUpddate", label: "时间" },
        { slot_name: "btn", label: "操作" },
      ],
      tableData: [],
      pagination: { page: 1, pageSize: 10, total: 0 },
    };
  },
  mounted() {
    this.getList();
  },
  methods: {
    getList() {
      let params = {
        page: this.pagination.page,
        limit: this.pagination.pageSize,
      };
      fetchList(params).then((res) => {
        this.tableData = res.datas;
        this.pagination.total = res.total;
      });
    },
    handlePageChange(val) {
      this.pagination = { ...this.pagination, ...val };
      this.getList();
    },
    show(row) {},
  },
};
</script>
