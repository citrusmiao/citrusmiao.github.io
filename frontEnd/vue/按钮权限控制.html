# 自定义指令实现按钮权限

  • 在 utils文件夹下新建 btnPermission.js 文件
import Vue from 'vue';
import store from '@/store'
/**
 * 操作按钮权限指令
 * 首先获取当前用户拥有的操作code列表
 * 在需要权限的按钮上添加属性 v-roles="`code`"   code为权限系统配置的操作code码
 */
const btnRoles = store.state.roleList

Vue.directive('roles', {
  mounted: ((el, binding) => {
    if (!btnRoles.includes(binding.value)) {
      el.parentNode.removeChild(el)
    }
  })
})
  • 在 main.js 中引入
import '@/utils/btnPermission' //按钮操作权限
  • 在 store state 中定义 roleList 数组
const state = {
  roleList: []
}

const mutations = {
  SET_ROLES: (state, rolesList) => {
    state.rolesList = rolesList
  }
}
  • 在按钮上添加指令，传入具体操作权限
 <el-button
    type="primary"
    icon="el-icon-plus"
    @click="func"
    v-roles="`drawUpIndex:create`">
    新增
    </el-button>
